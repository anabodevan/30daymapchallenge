Day 01 - Points

Challenge Classic: Map with point data (e.g., individual locations, points of interest, clusters). Focus on effective symbolization and density visualization.

For this one, we will be looking at the List of Largest LGBT Events on Wikipedia and map them by size.

SETUP

1.1 Load Libraries and Data

In order to get the data from Wikipedia to our R, we will use the rvest package

Code 
library(pacman)
pacman::p_load(tidyverse, ragg, rvest, rnaturalearth, showtext, 
              janitor, sf, tidygeocoder, ggrepel, scales)

events_raw <- read_html("https://en.wikipedia.org/wiki/List_of_largest_LGBT_events") |> 
  html_table() |> 
  purrr::pluck(4) |>  
  janitor::clean_names()

world <- ne_countries(scale = "medium", returnclass = "sf")

2. DATA WRANGLING

Now that we have the raw table, it is necessary to clean and transform the data.

Code 
events <- events_raw |> 
  mutate(
    rank = str_remove(rank, " \\(tie\\)"), # remove the "tie" events 
    # transform strings into numbers
    organizers_statistics = str_remove(organizers_statistics, "\\+*\\[.+"),
    organizers_statistics = parse_number(organizers_statistics),
    authorities_statistics = str_remove(authorities_statistics, "\\+*\\[.+"),
    authorities_statistics = parse_number(authorities_statistics),
    # creates a new column for attendence with organizer_statistics, or authorities if necessary
    attendance = if_else(!is.na(organizers_statistics), organizers_statistics, authorities_statistics)
  ) |> 
    # select the most recent event per city
  group_by(city) |> 
  slice_max(order_by = year, n = 1) |> 
  ungroup()

In order to correctly place the points in our map, we will need the coordinates for the cities present in the dataset. The tidygeocoder package will get this info for us automatically (a huge help! I used to depend on Google and AI to do this…)

Code 
events_geo <- events |> 
  slice_max(order_by = attendance, n = 10) |> # keep only the top 20 to avoid convolution 
  distinct(city, .keep_all = TRUE) |> 
  geocode(city = city, method = "osm")

# Convert to sf object 
events_geo_sf <- st_as_sf(events_geo, coords = c("long", "lat"), crs = 4326)
events_geo_sf <- events_geo_sf |>
  arrange(desc(attendance)) |>
  mutate(
    city_color = factor(city, levels = city)
  )

# Transform to Natural Earth projection
natearth_crs <- "+proj=natearth +lon_0=0 +datum=WGS84"
world_transformed <- st_transform(world, crs = natearth_crs)
events_transformed <- st_transform(events_geo_sf, crs = natearth_crs)

If all our data clean and ready, we can start the actual plotting.

3.2 Plot

Code 
ggplot() +
  # World map first
  geom_sf(data = world_transformed, fill = "black", color = "transparent", linewidth = 0.2) +
  
# Event points with pride gradient
  geom_sf(
    data = events_transformed,
    aes(size = attendance, fill = city_color),
    alpha = 0.9,
    shape = 21,
    color = "white",
    stroke = 1
  ) +
  
  # Discrete colors using pride palette
  scale_fill_manual(
    values = pal,
    labels = paste0(
    events_geo_sf |> arrange(desc(attendance)) |> pull(city),
    " - ",
    round((events_geo_sf |> arrange(desc(attendance)) |> pull(attendance)) / 1e6, 1),
    "M (",
    events_geo_sf |> arrange(desc(attendance)) |> pull(year),
    ")"
  ),
    name = NULL,
    guide = guide_legend(
    override.aes = list(size = 5),
    keywidth = 1,
    keyheight = 1
  )
) +
  
  scale_size_area(max_size = 10, guide = "none") + 
  
  # Natural Earth projection 
  coord_sf(datum = natearth_crs) +
  
  labs(
    title = title,
    subtitle = subtitle, 
    caption = caption
  ) + 
  
  theme_void(base_family = font_body) +
  theme(
    plot.title = element_text(
      size = 40, 
      face = "bold", 
      hjust = 0.5,
      margin = margin(t = 10, b = 5),
      family = font_title
    ),
    plot.subtitle = element_text(
      size = 20, 
      hjust = 0.5,
      margin = margin(b = 15),
      color = "#131212de"
    ),
    plot.caption = element_text(
      size = 12, 
      hjust = 0.5,
      margin = margin(t = 10, b = 5),
      color = "#666666"),

    plot.margin = margin(20, 20, 20, 20),
    plot.background = element_rect(fill = "white", color = NA),
    legend.position = "right",
    legend.text = element_text(size = 16, family = font_body),
    legend.spacing.y = unit(0.4, "cm")
  )
